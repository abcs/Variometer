244 /* This function is called by the library when in device mode, and must be overridden (see library "USB Descriptors"
245  * documentation) by the application code so that the address and size of a requested descriptor can be given
246  * to the USB library. When the device receives a Get Descriptor request on the control endpoint, this function
247  * is called so that the descriptor details can be passed back and the appropriate descriptor sent back to the
248  * USB host.
249  */
250 uint16_t CALLBACK_USB_GetDescriptor(uint8_t corenum,
251  const uint16_t wValue,
252  const uint8_t wIndex,
253  const void * *const DescriptorAddress) //uint8_t corenum,
254 {
255  const uint8_t DescriptorType = (wValue >> 8);
256  const uint8_t DescriptorNumber = (wValue & 0xFF);
257 
258  const void *Address = NULL;
259  uint16_t Size = NO_DESCRIPTOR;
260 
261  switch (DescriptorType) {
262  case DTYPE_Device:
263  Address = &DeviceDescriptor;
264  Size = sizeof(USB_Descriptor_Device_t);
265  break;
266 
267  case DTYPE_Configuration:
268  Address = &ConfigurationDescriptor;
269  Size = ConfigurationDescriptor.Config.TotalConfigurationSize;
270  break;
271 
272  case DTYPE_String:
273  switch (DescriptorNumber) {
274  case 0x00:
275  Address = LanguageStringPtr;
276  Size = pgm_read_byte(&LanguageStringPtr->Header.Size);
277  break;
278 
279  case 0x01:
280  Address = ManufacturerStringPtr;
281  Size = pgm_read_byte(&ManufacturerStringPtr->Header.Size);
282  break;
283 
284  case 0x02:
285  Address = ProductStringPtr;
286  Size = pgm_read_byte(&ProductStringPtr->Header.Size);
287  break;
288  }
289 
290  break;
291  }
292 
293  *DescriptorAddress = Address;
294  return Size;
295 }
